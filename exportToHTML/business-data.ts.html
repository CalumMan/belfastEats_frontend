1    // src/app/services/business-data.ts
2    import { Injectable } from '@angular/core';
3    import { HttpClient } from '@angular/common/http';
4    import { Observable, of } from 'rxjs';
5    import { map, catchError, tap } from 'rxjs/operators';
6    
7    import localData from '../../assets/businesses_with_coords.json';
8    
9    @Injectable({
10     providedIn: 'root',
11   })
12   export class BusinessData {
13     // NOTE: trailing slash to match Flask route
14     private apiUrl = 'http://127.0.0.1:5000/api/v1.0/restaurants/';
15   
16     private businesses: any[] = [];
17     private source: 'api' | 'json' | 'none' = 'none';
18     pageSize: number = 3;
19   
20     constructor(private http: HttpClient) {}
21   
22     /**
23      * Load businesses from the API (first time) and cache them. 
24      * If the API fails, fall back to local JSON. 
25      */
26     loadBusinesses(): Observable<any[]> {
27       // If we already have data, just return it
28       if (this.businesses.length) {
29         return of(this.businesses);
30       }
31   
32       return this.http.get<any[]>(this.apiUrl).pipe(
33         map((apiData) => {
34           console.log(
35             '[BusinessData] Loaded restaurants from API:',
36             apiData.length
37           );
38   
39     const normalised = apiData.map((r: any) => ({
40     _id: r._id ?? r.id ?? r.restaurant_id,
41     name: r.name ?? r.BusinessName ?? 'Unknown name',
42     address: r.address ?? r.AddressLine1 ?? '',
43     postcode: r.postcode ?? r.PostCode ?? '',
44     hygiene_rating: r.hygiene_rating ?? r.HygieneRating ?? r.rating ?? null,
45     cuisine: r.cuisine ?? r.BusinessType ?? '',
46     tags: r.tags ?? [],
47   
48     // ⭐ NEW — coordinates from API (or fallback)
49     lat: r.latitude ?? r.lat ?? null,
50     lng: r.longitude ?? r.lng ?? null,
51   
52     ...r, // keep all original fields
53   }));
54   
55   
56           this.businesses = normalised;
57           this.source = 'api';
58           return this.businesses;
59         }),
60         catchError((err) => {
61           console.warn(
62             '[BusinessData] API unreachable, using local JSON instead.',
63             err
64           );
65           this.businesses = localData as any[];
66           this.source = 'json';
67           return of(this.businesses);
68         })
69       );
70     }
71   
72     getSource() {
73       return this.source;
74     }
75   
76     getAllBusinesses() {
77       return this.businesses;
78     }
79   
80     getStats() {
81       return this.http.get<{ restaurants_count: number; reviews_count: number; average_rating: number | null }>(
82         `${this.apiUrl}stats`
83       );
84     }
85   
86     // Admin: create new business
87     createBusiness(payload: any): Observable<any> {
88       return this.http.post<any>(this.apiUrl, payload).pipe(
89         tap((created) => {
90           if (created) {
91             this.businesses.unshift(created);
92           }
93         })
94       );
95     }
96   
97     // Admin: update business
98     updateBusiness(id: string, payload: any): Observable<any> {
99       return this.http.put<any>(`${this.apiUrl}${id}`, payload).pipe(
100        tap((updated) => {
101          if (!updated) return;
102          const idx = this.businesses.findIndex((b) => b._id === id);
103          if (idx >= 0) {
104            this.businesses[idx] = { ...this.businesses[idx], ...updated };
105          }
106        })
107      );
108    }
109  
110    // Admin: delete business
111    deleteBusiness(id: string): Observable<any> {
112      return this.http.delete(`${this.apiUrl}${id}`).pipe(
113        tap(() => {
114          this.businesses = this.businesses.filter((b) => b._id !== id);
115        })
116      );
117    }
118  
119    getBusiness(id: string) {
120      return this.businesses.filter((b) => b._id === id);
121    }
122  
123    getBusinesses(page: number) {
124      const start = (page - 1) * this.pageSize;
125      const end = start + this.pageSize;
126      return this.businesses.slice(start, end);
127    }
128  
129    getLastPageNumber() {
130      return Math.ceil(this.businesses.length / this.pageSize);
131    }
132  }
