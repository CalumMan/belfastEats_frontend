1    import { Injectable } from "@angular/core";
2    import { HttpClient } from "@angular/common/http";
3    import { Observable, map, catchError, of } from "rxjs";
4    
5    type LatLngLiteral = { lat: number; lng: number };
6    
7    // Response shape from api.postcodes.io
8    interface PostcodeApiResponse {
9      status: number;
10     result: {
11       latitude: number;
12       longitude: number;
13     } | null;
14   }
15   
16   @Injectable({
17     providedIn: "root",
18   })
19   export class GeoService {
20     private apiUrl = "https://api.postcodes.io/postcodes";
21   
22     constructor(private http: HttpClient) {}
23   
24     geocode(postcode: string): Observable<LatLngLiteral | null> {
25       const cleaned = postcode.trim();
26   
27       console.log("[GeoService] Geocoding postcode:", cleaned);
28   
29       return this.http
30         .get<PostcodeApiResponse>(
31           `${this.apiUrl}/${encodeURIComponent(cleaned)}`
32         )
33         .pipe(
34           map((res) => {
35             if (res.status !== 200 || !res.result) {
36               console.warn("[GeoService] No result for postcode", cleaned);
37               return null;
38             }
39   
40             return {
41               lat: res.result.latitude,
42               lng: res.result.longitude,
43             };
44           }),
45           catchError((err) => {
46             console.error("[GeoService] Error calling postcodes.io", err);
47             return of(null);
48           })
49         );
50     }
51   }
