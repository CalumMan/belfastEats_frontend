1    import { Component } from '@angular/core';
2    import { CommonModule } from '@angular/common';
3    import {
4      FormBuilder,
5      FormGroup,
6      ReactiveFormsModule,
7      Validators,
8    } from '@angular/forms';
9    import { Router } from '@angular/router';
10   import { AuthService } from '../../services/auth.service';
11   
12   @Component({
13     selector: 'app-login',
14     standalone: true,
15     imports: [CommonModule, ReactiveFormsModule],
16     templateUrl: './login.html',
17   })
18   export class Login {
19     form: FormGroup;
20     loading = false;
21     error: string | null = null;
22   
23     constructor(
24       private fb: FormBuilder,
25       private auth: AuthService,
26       private router: Router
27     ) {
28       this.form = this.fb.group({
29         email: ['', [Validators.required, Validators.email]],
30         password: ['', [Validators.required, Validators.minLength(6)]],
31       });
32     }
33   
34     get email() {
35       return this.form.get('email');
36     }
37   
38     get password() {
39       return this.form.get('password');
40     }
41   
42     onSubmit() {
43       if (this.form.invalid) {
44         this.form.markAllAsTouched();
45         return;
46       }
47   
48       this.loading = true;
49       this.error = null;
50   
51       const { email, password } = this.form.value;
52   
53       this.auth.login(email, password).subscribe({
54         next: () => {
55           this.loading = false;
56           this.router.navigate(['/businesses']);
57         },
58         error: (err) => {
59           this.loading = false;
60           this.error =
61             err?.error?.msg ||
62             err?.error?.error ||
63             'Login failed. Please check your credentials.';
64           console.error('Login error', err);
65         },
66       });
67     }
68   }
