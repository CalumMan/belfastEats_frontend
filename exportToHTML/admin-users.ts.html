admin-users.ts 
1    import { Component, OnInit } from '@angular/core';
2    import { CommonModule } from '@angular/common';
3    import { ReactiveFormsModule, FormBuilder, FormGroup, Validators } from '@angular/forms';
4    import { UsersDebugService, DebugUser } from '../../services/users-debug.service';
5    import { AuthService } from '../../services/auth.service';
6    import { RouterModule } from '@angular/router';
7    
8    @Component({
9      selector: 'app-admin-users',
10     standalone: true,
11     imports: [CommonModule, ReactiveFormsModule, RouterModule],
12     templateUrl: './admin-users.html',
13   })
14   export class AdminUsers implements OnInit {
15     users: DebugUser[] = [];
16     editing: string | null = null;
17     form!: FormGroup;
18     info: string | null = null;
19   
20     constructor(
21       private usersService: UsersDebugService,
22       public auth: AuthService,
23       private fb: FormBuilder
24     ) {}
25   
26     ngOnInit(): void {
27       this.form = this.fb.group({
28         username: ['', [Validators.required]],
29         email: ['', [Validators.required, Validators.email]],
30         role: ['user', [Validators.required]],
31       });
32       this.load();
33     }
34   
35     load() {
36       this.users = this.usersService.list();
37     }
38   
39     startEdit(user: DebugUser) {
40       this.editing = user.id;
41       this.form.setValue({
42         username: user.username,
43         email: user.email,
44         role: user.role,
45       });
46       this.info = null;
47     }
48   
49     cancel() {
50       this.editing = null;
51       this.form.reset({
52         username: '',
53         email: '',
54         role: 'user',
55       });
56     }
57   
58     save() {
59       if (!this.editing || this.form.invalid) {
60         this.form.markAllAsTouched();
61         return;
62       }
63       const updated = this.usersService.update(this.editing, this.form.value);
64       this.info = updated ? 'User updated (debug only, local data)' : 'User not found';
65       this.cancel();
66       this.load();
67     }
68   
69     remove(id: string) {
70       this.usersService.delete(id);
71       this.load();
72     }
73   }
