1    import { Injectable } from '@angular/core';
2    import { HttpClient } from '@angular/common/http';
3    import { Observable, map } from 'rxjs';
4    
5    // Public shape used by the UI
6    export interface Review {
7      id: string;
8      businessId: string;
9      author: string;
10     rating: number;
11     comment: string;
12     createdAt: string;
13     title?: string;
14     userId?: string;
15     restaurantName?: string;
16   }
17   
18   // API response shape from Flask
19   interface ApiReview {
20     _id: string;
21     restaurant_id: string;
22     user_id: string;
23     user_name?: string;
24     restaurant_name?: string;
25     rating: number;
26     title?: string;
27     body?: string;
28     created_at?: string;
29     updated_at?: string;
30   }
31   
32   @Injectable({
33     providedIn: 'root',
34   })
35   export class ReviewData {
36     private apiUrl = 'http://127.0.0.1:5000/api/v1.0/reviews';
37   
38     constructor(private http: HttpClient) {}
39   
40     getReviewsForBusiness(businessId: string): Observable<Review[]> {
41       return this.http
42         .get<ApiReview[]>(`${this.apiUrl}/restaurant/${businessId}`)
43         .pipe(map((list) => list.map((r) => this.mapReview(r))));
44     }
45   
46     addReview(
47       businessId: string,
48       rating: number,
49       comment: string,
50       title?: string
51     ): Observable<Review> {
52       return this.http
53         .post<ApiReview>(`${this.apiUrl}/${businessId}`, {
54           rating,
55           title: title ?? '',
56           body: comment ?? '',
57         })
58         .pipe(map((r) => this.mapReview(r)));
59     }
60   
61     updateReview(
62       reviewId: string,
63       rating: number,
64       comment: string,
65       title?: string
66     ): Observable<Partial<Review>> {
67       return this.http.put(`${this.apiUrl}/${reviewId}`, {
68         rating,
69         body: comment,
70         title,
71       }).pipe(map(() => ({
72         rating,
73         comment,
74         title,
75       })));
76     }
77   
78     deleteReview(reviewId: string): Observable<any> {
79       return this.http.delete(`${this.apiUrl}/${reviewId}`);
80     }
81   
82     getMyReviews(): Observable<Review[]> {
83       return this.http
84         .get<ApiReview[]>(`${this.apiUrl}/me`)
85         .pipe(map((list) => list.map((r) => this.mapReview(r))));
86     }
87   
88     private mapReview(api: ApiReview): Review {
89       const created =
90         api.updated_at ??
91         api.created_at ??
92         new Date().toISOString();
93   
94       return {
95         id: api._id,
96         businessId: api.restaurant_id,
97         author: api.user_name ?? api.user_id ?? 'Unknown user',
98         restaurantName: api.restaurant_name,
99         userId: api.user_id,
100        rating: api.rating,
101        title: api.title ?? '',
102        comment: api.body ?? api.title ?? '',
103        createdAt: created,
104      };
105    }
106  }
