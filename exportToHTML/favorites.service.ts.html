1    import { Injectable } from '@angular/core';
2    
3    export interface FavoriteBusiness {
4      id: string;
5      name: string;
6      address?: string;
7      postcode?: string;
8      hygiene_rating?: number | null;
9      addedAt: string;
10   }
11   
12   @Injectable({ providedIn: 'root' })
13   export class FavoritesService {
14     private storageKey = 'be_favorites';
15   
16     private read(): FavoriteBusiness[] {
17       const raw = localStorage.getItem(this.storageKey);
18       if (!raw) return [];
19       try {
20         return JSON.parse(raw) as FavoriteBusiness[];
21       } catch {
22         return [];
23       }
24     }
25   
26     private write(list: FavoriteBusiness[]) {
27       localStorage.setItem(this.storageKey, JSON.stringify(list));
28     }
29   
30     getAll(): FavoriteBusiness[] {
31       return this.read();
32     }
33   
34     isFavorite(id: string): boolean {
35       return this.read().some((f) => f.id === id);
36     }
37   
38     add(fav: Omit<FavoriteBusiness, 'addedAt'>): FavoriteBusiness {
39       const current = this.read();
40       if (current.some((f) => f.id === fav.id)) {
41         return current.find((f) => f.id === fav.id)!;
42       }
43       const newFav: FavoriteBusiness = { ...fav, addedAt: new Date().toISOString() };
44       this.write([newFav, ...current]);
45       return newFav;
46     }
47   
48     remove(id: string) {
49       const filtered = this.read().filter((f) => f.id !== id);
50       this.write(filtered);
51     }
52   }
