1    import { TestBed } from '@angular/core/testing';
2    import {
3      HttpClientTestingModule,
4      HttpTestingController,
5    } from '@angular/common/http/testing';
6    import { BusinessData } from './business-data';
7    
8    describe('BusinessData', () => {
9      let service: BusinessData;
10     let http: HttpTestingController;
11   
12     const apiUrl = 'http://127.0.0.1:5000/api/v1.0/restaurants/';
13   
14     beforeEach(() => {
15       TestBed.configureTestingModule({
16         imports: [HttpClientTestingModule],
17         providers: [BusinessData],
18       });
19   
20       service = TestBed.inject(BusinessData);
21       http = TestBed.inject(HttpTestingController);
22     });
23   
24     afterEach(() => {
25       http.verify();
26     });
27   
28     it('loads businesses from API on success', () => {
29       const mock = [
30         { _id: '1', name: 'Alpha', address: 'A', postcode: 'BT1' },
31         { _id: '2', name: 'Beta', address: 'B', postcode: 'BT2' },
32       ];
33       let result: any[] | undefined;
34   
35       service.loadBusinesses().subscribe((data) => {
36         result = data;
37         expect(data.length).toBe(mock.length);
38         expect(data[0].name).toBe('Alpha');
39         expect(service.getSource()).toBe('api');
40       });
41   
42       const req = http.expectOne(apiUrl);
43       expect(req.request.method).toBe('GET');
44       req.flush(mock);
45   
46       expect(result).toBeDefined();
47     });
48   
49     it('falls back to local JSON when API fails', () => {
50       let result: any[] | undefined;
51   
52       service.loadBusinesses().subscribe((data) => {
53         result = data;
54         expect(data.length).toBeGreaterThan(0);
55         expect(service.getSource()).toBe('json');
56       });
57   
58       const req = http.expectOne(apiUrl);
59       req.error(new ProgressEvent('error'));
60   
61       expect(result).toBeDefined();
62     });
63   
64     it('paginates businesses correctly', () => {
65       const mock = [
66         { _id: '1', name: 'One' },
67         { _id: '2', name: 'Two' },
68         { _id: '3', name: 'Three' },
69         { _id: '4', name: 'Four' },
70         { _id: '5', name: 'Five' },
71       ];
72   
73       // seed cache directly
74       (service as any).businesses = mock;
75       service.pageSize = 2;
76   
77       const page1 = service.getBusinesses(1);
78       const page2 = service.getBusinesses(2);
79       const page3 = service.getBusinesses(3);
80   
81       expect(page1.map((b) => b._id)).toEqual(['1', '2']);
82       expect(page2.map((b) => b._id)).toEqual(['3', '4']);
83       expect(page3.map((b) => b._id)).toEqual(['5']);
84       expect(service.getLastPageNumber()).toBe(3);
85     });
86   });
