1    import { TestBed } from '@angular/core/testing';
2    import { ReactiveFormsModule } from '@angular/forms';
3    import { RouterTestingModule } from '@angular/router/testing';
4    import { of } from 'rxjs';
5    
6    import { Login } from './login';
7    import { AuthService } from '../../services/auth.service';
8    
9    describe('Login component', () => {
10     let authMock: { login: (email: string, password: string) => any; calls: any[] };
11   
12     beforeEach(async () => {
13       authMock = {
14         calls: [],
15         login: (email: string, password: string) => {
16           authMock.calls.push([email, password]);
17           return of(void 0);
18         },
19       };
20   
21       await TestBed.configureTestingModule({
22         imports: [Login, ReactiveFormsModule, RouterTestingModule],
23         providers: [{ provide: AuthService, useValue: authMock }],
24       }).compileComponents();
25     });
26   
27     it('shows validation errors when submitted empty', () => {
28       const fixture = TestBed.createComponent(Login);
29       const component = fixture.componentInstance;
30   
31       component.onSubmit();
32       fixture.detectChanges();
33   
34       expect(component.form.invalid).toBe(true);
35       const text = fixture.nativeElement.textContent as string;
36       expect(text).toContain('Valid email is required');
37       expect(text).toContain('Password is required');
38     });
39   
40     it('calls AuthService.login with form values', () => {
41       const fixture = TestBed.createComponent(Login);
42       const component = fixture.componentInstance;
43   
44       component.form.setValue({
45         email: 'test@example.com',
46         password: 'secret1',
47       });
48   
49       component.onSubmit();
50   
51       expect(authMock.calls.length).toBe(1);
52       expect(authMock.calls[0]).toEqual(['test@example.com', 'secret1']);
53     });
54   });
