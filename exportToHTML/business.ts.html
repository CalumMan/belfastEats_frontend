24   @Component({

26     standalone: true,
27     imports: [CommonModule, ReactiveFormsModule, RouterModule],

31   export class Business implements AfterViewInit {
32     @ViewChild('mapContainer') mapContainer?: ElementRef<HTMLDivElement>;
33     private viewReady = false;
34     private pendingCenter: LatLngLiteral | null = null;
35     private mapInitAttempts = 0;
36   
37     business_list: any[] = [];
38   
39     private readonly fallbackCenter: LatLngLiteral = {
40       lat: 54.5973,
41       lng: -5.9301,
42     };
43   
44     map_options: MapOptions = {
45       center: { ...this.fallbackCenter },
46       zoom: 13,
47     };
48     marker_position: LatLngLiteral | null = null;
49   
50     // Reviews
51     reviews: Review[] = [];
52     reviewForm!: FormGroup;
53     submittingReview = false;
54     loadingReviews = false;
55     reviewsError: string | null = null;
56     favMessage: string | null = null;
57     isFavorite = false;
58   
59     businessId: string | null = null;

61     constructor(
62       private businessData: BusinessData,
63       private reviewData: ReviewData,
64       private route: ActivatedRoute,
65       private fb: FormBuilder,
66       public auth: AuthService,
67       private cdr: ChangeDetectorRef,
68       private favorites: FavoritesService

71     ngAfterViewInit(): void {
72       this.viewReady = true;
73       if (this.pendingCenter) {
74         this.renderMap(this.pendingCenter.lat, this.pendingCenter.lng);
75         this.pendingCenter = null;
76       }
77     }

79     ngOnInit() {
80       // React to route changes
81       this.route.paramMap.subscribe((params) => {
82         this.businessId = params.get('id');
83   
84         if (!this.businessId) {
85           console.warn('[Business] No id in route');
86           return;
87         }
88   
89         // reset map while loading new business
90         this.marker_position = null;
91         this.map_options = {
92           center: { ...this.fallbackCenter },
93           zoom: 13,
94         };
95   
96         // try to get from already loaded data
97         this.business_list = this.businessData.getBusiness(this.businessId);
98   
99         if (!this.business_list.length) {
100          // fallback: load all then pick one
101          this.businessData.loadBusinesses().subscribe(() => {
102            this.business_list = this.businessData.getBusiness(this.businessId!);
103            this.setupBusinessAndMap();
104          });
105        } else {
106          this.setupBusinessAndMap();
107        }
108  
109        // refresh reviews for this business
110        if (this.businessId) {
111          this.loadReviews(this.businessId);
112        }
113      });
114  
115      // build review form once
116      this.reviewForm = this.fb.group({
117        rating: [5, [Validators.required, Validators.min(1), Validators.max(5)]],
118        comment: ['', [Validators.required, Validators.maxLength(500)]],
119        title: ['', [Validators.maxLength(120)]],
120      });
121    }
122  
123    private setupBusinessAndMap() {
124      const business = this.business_list[0];
125      if (!business) {
126        console.warn('[Business] No business found for id', this.businessId);
127        return;
128      }
129      this.isFavorite = !!(business._id && this.favorites.isFavorite(business._id));
130  
131      // Prefer normalised lat/lng, but also support raw latitude/longitude from API/DB
132      const lat = Number(
133        business.lat ??
134        business.latitude ??
135        54.5973
136      );
137      const lng = Number(
138        business.lng ??
139        business.longitude ??
140        -5.9301
141      );
142  
143      console.log('[MAP]', {
144        id: this.businessId,
145        name: business.name,
146        lat,
147        lng,
148      });
149  
150      this.map_options = {
151        center: { lat, lng },
152        zoom: 15,
153      };
154  
155      this.marker_position = { lat, lng };
156  
157      console.log('[MAP] set marker_position', this.marker_position);
158  
159      this.renderMap(lat, lng);
160  
161      if (this.businessId) {
162        this.loadReviews(this.businessId);
163      }
164    }
165  
166    addFavorite() {
167      if (!this.businessId || this.auth.isAdmin()) return;
168      const biz = this.business_list[0];
169      if (!biz) return;
170      this.favorites.add({
171        id: this.businessId,
172        name: biz.name || 'Unnamed business',
173        address: biz.address,
174        postcode: biz.postcode,
175        hygiene_rating: biz.hygiene_rating,
176      });
177      this.isFavorite = true;
178      this.favMessage = 'Added to favourites';
179    }
180  
181    private loadReviews(businessId: string) {
182      this.loadingReviews = true;
183      this.reviewsError = null;
184  
185      console.log('[Reviews] loading for business', businessId);
186  
187      this.reviewData
188        .getReviewsForBusiness(businessId)
189        .pipe(
190          finalize(() => (this.loadingReviews = false))
191        )
192        .subscribe({
193          next: (reviews) => {
194            const currentUserId = this.auth.getCurrentUser()?.id;
195            this.reviews = reviews
196              .map((r) => ({
197                ...r,
198                author: currentUserId && r.userId === currentUserId ? 'You' : (r.author || 'Anonymous'),
199                title: r.title ?? '',
200              }))
201              .sort(
202                (a, b) =>
203                  new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime()
204              );
205  
206            console.log('[Reviews] loaded', this.reviews.length);
207            this.cdr.detectChanges();
208          },
209          error: (err) => {
210            this.reviewsError =
211              err?.error?.error || 'Could not load reviews right now.';
212            console.error('[Reviews] load error', err);
213            this.reviews = [];
214            this.cdr.detectChanges();
215          },
216        });
217    }
218  
219    private renderMap(lat: number, lng: number) {
220      if (!this.viewReady) {
221        this.pendingCenter = { lat, lng };
222        return;
223      }
224      if (!this.mapContainer?.nativeElement) {
225        console.warn('[MAP] mapContainer missing');
226        return;
227      }
228      if (!(window as any)?.google?.maps) {
229        this.mapInitAttempts++;
230        if (this.mapInitAttempts <= 20) {
231          // retry until script loads (up to ~5s)
232          setTimeout(() => this.renderMap(lat, lng), 250);
233        } else {
234          console.warn('[MAP] google maps script not loaded after retries');
235        }
236        return;
237      }
238  
239      this.mapInitAttempts = 0;
240  
241      const map = new google.maps.Map(this.mapContainer.nativeElement, {
242        center: { lat, lng },
243        zoom: 15,
244      });
245  
246      new google.maps.Marker({
247        position: { lat, lng },
248        map,
249      });
250    }

265    onSubmitReview() {
266      if (!this.businessId || this.reviewForm.invalid) {
267        this.reviewForm.markAllAsTouched();
268        return;
269      }
270  
271      this.submittingReview = true;
272  
273      const { rating, comment, title } = this.reviewForm.value;
274  
275      this.reviewData
276        .addReview(this.businessId, rating, comment, title)
277        .subscribe({
278          next: () => {
279            this.submittingReview = false;
280            this.reviewForm.reset({
281              rating: 5,
282              comment: '',
283              title: '',
284            });
285            this.loadReviews(this.businessId!);
286          },
287          error: (err) => {
288            this.submittingReview = false;
289            this.reviewsError =
290              err?.error?.error ||
291              'Could not submit your review. Please try again.';
292            console.error('[Reviews] submit error', err);
293          },
294        });
295    }
