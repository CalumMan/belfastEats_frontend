1    import { Component, HostListener, OnInit } from '@angular/core';
2    import { ActivatedRoute, RouterLink } from '@angular/router';
3    import { CommonModule } from '@angular/common';
4    import { FormsModule } from '@angular/forms';
5    import { BusinessData } from '../../services/business-data';
6    
7    @Component({
8      selector: 'app-businesses',
9      standalone: true,
10     providers: [],
11     imports: [CommonModule, FormsModule, RouterLink],
12     templateUrl: './businesses.html',
13     styleUrl: './businesses.css',
14   })
15   export class Businesses implements OnInit {
16     // All businesses (after API/JSON load)
17     all_businesses: any[] = [];
18   
19     // After search/filter/sort, before paging
20     filtered_businesses: any[] = [];
21   
22     // The slice for the current page (what the template renders)
23     business_page: any[] = [];
24   
25     // Pagination
26     page: number = 1;
27     pageSize: number = 6;
28     totalPages: number = 1;
29   
30     // Search / filter / sort state
31     searchTerm: string = '';
32     minRating: number = 0;
33     sortOption: string = 'name-asc';
34   
35     // Data source flag for the badge: 'api' | 'json' | 'none'
36     dataSource: 'api' | 'json' | 'none' = 'none';
37     private initialMinRating: number | null = null;
38   
39     constructor(public businessData: BusinessData, private route: ActivatedRoute) {}
40   
41     ngOnInit(): void {
42       this.page = 1;
43       this.setPageSizeForViewport();
44   
45       this.route.queryParamMap.subscribe((params) => {
46         const min = Number(params.get('minRating'));
47         if (!isNaN(min) && min > 0) {
48           this.initialMinRating = min;
49         }
50       });
51   
52       this.dataSource = 'none';
53   
54       // Load from API (or JSON fallback) then apply filters
55       this.businessData.loadBusinesses().subscribe((data) => {
56         this.all_businesses = data;
57         this.dataSource = this.businessData.getSource();
58   
59         if (this.initialMinRating !== null) {
60           this.minRating = this.initialMinRating;
61         }
62   
63         console.log(
64           '[Businesses] loaded',
65           this.all_businesses.length,
66           'records from',
67           this.dataSource
68         );
69   
70         this.applyFilters(true);
71       });
72     }
73   
74     @HostListener('window:resize')
75     onResize() {
76       const prev = this.pageSize;
77       this.setPageSizeForViewport();
78       if (this.pageSize !== prev) {
79         this.applyFilters(true);
80       }
81     }
82   
83     private setPageSizeForViewport() {
84       const width = window.innerWidth;
85       if (width >= 1200) {
86         this.pageSize = 6; // show 6 cards on wide screens
87       } else {
88         this.pageSize = 3; // show 3 cards on narrower screens
89       }
90     }
91   
92     applyFilters(resetPage: boolean = true): void {
93       let results = [...this.all_businesses];
94   
95       const term = this.searchTerm.trim().toLowerCase();
96       if (term.length > 0) {
97         results = results.filter((biz: any) =>
98           (biz.name ?? '').toLowerCase().includes(term) ||
99           (biz.address ?? '').toLowerCase().includes(term) ||
100          (biz.postcode ?? '').toLowerCase().includes(term)
101        );
102      }
103  
104      if (this.minRating > 0) {
105        results = results.filter(
106          (biz: any) => (biz.hygiene_rating ?? 0) >= this.minRating
107        );
108      }
109  
110      switch (this.sortOption) {
111        case 'name-asc':
112          results.sort((a, b) => (a.name ?? '').localeCompare(b.name ?? ''));
113          break;
114        case 'name-desc':
115          results.sort((a, b) => (b.name ?? '').localeCompare(a.name ?? ''));
116          break;
117        case 'rating-desc':
118          results.sort(
119            (a, b) => (b.hygiene_rating ?? 0) - (a.hygiene_rating ?? 0)
120          );
121          break;
122        case 'rating-asc':
123          results.sort(
124            (a, b) => (a.hygiene_rating ?? 0) - (b.hygiene_rating ?? 0)
125          );
126          break;
127      }
128  
129      this.filtered_businesses = results;
130  
131      this.totalPages = Math.max(
132        1,
133        Math.ceil(this.filtered_businesses.length / this.pageSize)
134      );
135  
136      if (resetPage || this.page > this.totalPages) {
137        this.page = 1;
138      }
139  
140      this.updatePageSlice();
141    }
142  
143    private updatePageSlice(): void {
144      const start = (this.page - 1) * this.pageSize;
145      const end = start + this.pageSize;
146      this.business_page = this.filtered_businesses.slice(start, end);
147      sessionStorage['page'] = String(this.page);
148    }
149  
150    previousPage(): void {
151      if (this.page > 1) {
152        this.page = this.page - 1;
153        this.updatePageSlice();
154      }
155    }
156  
157    nextPage(): void {
158      if (this.page < this.totalPages) {
159        this.page = this.page + 1;
160        this.updatePageSlice();
161      }
162    }

164    trackById(index: number, item: any) {
165      return item?._id || index;
166    }

168    ratingStars(biz: any): any[] {
169      const n = Math.max(0, Math.min(5, Number(biz?.hygiene_rating) || 0));
170      return Array.from({ length: n });
171    }
